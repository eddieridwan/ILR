 58:   def calc_shipping_cost(cart)
 59:     if self.shipping_method.nil? || self.shipping_method.empty?
 60:       return nil
 61:     end
 62:     # Assume that products are limited to CDs
 63:     # Later have to add weight for each product
 64:     cart_weight = cart.total_items * 125 + 100 # in grams, incl packaging
 65:     case self.shipping_method
 66:     when "d-std"
 67:       # Use Australia Post regular parcel tables (Mar 2009)
 68:       case cart_weight
 69:       # Using prepaid parcel post satchel
 70:       when 000..500 then 5.50 + 2.50 
 71:       when 501..3000 then 9.60 + 4.00
 72:       else
 73:         # Above 500g (without satchel), Australia Post charges are based on distance
 74:         # The value used here is towards the upper range charge per part of kg
 75:         # From Canberra, range is $0.50 to $3.10 per part kg
 76:         9.05 + (Float(cart_weight) / Float(1000)).ceil * 2.50 + 5 
 77:       end
 78:     when "d-exp"
 79:       # Use Australia Post express parcel tables (Mar 2009)
 80:       case cart_weight
 81:       # Fixed price up to 500gms
 82:       when 000..500 then 7.70 + 2.50 
 83:       else
 84:         # Above 500g, Australia Post charges are based on distance
 85:         # The value used here is towards the upper range charge per part of kg
 86:         # From Canberra, range is $2.20 to $18.00 per part kg
 87:         15.70 + (Float(cart_weight) / Float(1000)).ceil * 12.00 + 5 
 88:       end
 89:     when "i-air"
 90:       # Use Australia Post international air parcel tables (Mar 2009)
 91:       case cart_weight
 92:       # Rate is determined by zone of destination country
 93:       when 000..250 then INTERNATIONAL_AIRMAIL[0][ship_to_zone(self.country)] + 5
 94:       when 251..500 then INTERNATIONAL_AIRMAIL[1][ship_to_zone(self.country)] + 5
 95:       when 501..750 then INTERNATIONAL_AIRMAIL[2][ship_to_zone(self.country)] + 5
 96:       when 751..1000 then INTERNATIONAL_AIRMAIL[3][ship_to_zone(self.country)] + 5
 97:       when 1001..1250 then INTERNATIONAL_AIRMAIL[4][ship_to_zone(self.country)] + 5
 98:       when 1251..1500 then INTERNATIONAL_AIRMAIL[5][ship_to_zone(self.country)] + 5
 99:       when 1501..1750 then INTERNATIONAL_AIRMAIL[6][ship_to_zone(self.country)] + 5
100:       when 1751..2000 then INTERNATIONAL_AIRMAIL[7][ship_to_zone(self.country)] + 5
101:       else
102:         # Above 2000g, there is an incremental charge for extra 500g or part of
103:         INTERNATIONAL_AIRMAIL[7][ship_to_zone] + 
104:           (Float(cart_weight-2000) / Float(500)).ceil * INTERNATIONAL_AIRMAIL_INC[ship_to_zone(self.country)] + 5 
105:       end
106:     when "i-courier"
107:       # Use Australia Post Express Courier International air parcel tables (Mar 2009)
108:       case cart_weight
109:       # Rate is determined by zone of destination country
110:       when 000..500 then INTERNATIONAL_COURIER[0][ship_to_zone(self.country)] + 5
111:       when 501..1000 then INTERNATIONAL_COURIER[1][ship_to_zone(self.country)] + 5
112:       when 1001..2000 then INTERNATIONAL_COURIER[2][ship_to_zone(self.country)] + 5
113:       when 2001..3000 then INTERNATIONAL_COURIER[3][ship_to_zone(self.country)] + 5
114:       else
115:         # Above 3000g, raise an exception as unlikely to happen
116:         raise "Maximum for shipping by International Courier is 3kg"
117:       end
118:     else
119:       logger.error "Invalid shipping method: " + self.shipping_method
120:       raise "Invalid shipping method"
121:     end
122:   end
